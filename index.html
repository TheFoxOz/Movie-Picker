`; document.body.appendChild(nav); return nav; } renderNavButton(tabName, icon, label) { const isActive = this.currentTab === tabName; return ` `; } setupNavigationListeners() { this.bottomNav.addEventListener('click', (e) => { const btn = e.target.closest('.nav-btn'); if (btn) this.navigateToTab(btn.dataset.tab); }); } navigateToTab(tabName) { if (!tabName) return; this.currentTab = tabName; if (tabName === 'swipe') this.tabs.swipe = new SwipeTab(); this.updateNavigation(); this.renderCurrentTab(); } updateNavigation() { this.bottomNav.querySelectorAll('.nav-btn').forEach(btn => { const isActive = btn.dataset.tab === this.currentTab; btn.style.background = isActive ?
